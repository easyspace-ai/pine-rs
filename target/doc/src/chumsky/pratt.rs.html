1//! Utilities for parsing expressions using
2//! [Pratt parsing](https://en.wikipedia.org/wiki/Operator-precedence_parser#Pratt_parsing).
3//!
4//! *“Who am I? What is my purpose in life? Does it really, cosmically speaking, matter if I don’t get up and go to work?”*
5//!
6//! Pratt parsing is a powerful technique for defining and parsing operators of varying arity, precedence, and
7//! associativity. Unlike [precedence climbing](https://en.wikipedia.org/wiki/Operator-precedence_parser), which
8//! defines operator precedence by structurally composing parsers of decreasing precedence, Pratt parsing defines
9//! precedence through a numerical
10//! ['binding power'](https://matklad.github.io/2020/04/13/simple-but-powerful-pratt-parsing.html#From-Precedence-to-Binding-Power)
11//! that determines how strongly operators should bind to the operands around them.
12//!
13//! Pratt parsers are defined with the [`Parser::pratt`] method.
14//!
15//! When writing pratt parsers, it is necessary to first define an 'atomic' operand used by the parser for building up
16//! expressions. In most languages, atoms are simple, self-delimiting patterns such as numeric and string literals,
17//! identifiers, or parenthesised expressions. Once an atom has been defined, operators can also be defined that
18//! operate upon said atoms.
19//!
20//! # Fold functions
21//!
22//! Because operators bind atoms together, pratt parsers require you to specify, for each operator, a function that
23//! combines its operands together into a syntax tree. These functions are given as the last arguments of [`infix`],
24//! [`prefix`], and [`postfix`].
25//!
26//! # Examples
27//!
28//! ```
29//! use chumsky::prelude::*;
30//! use chumsky::pratt::*;
31//! use chumsky::extra;
32//!
33//! enum Expr {
34//!     Add(Box<Self>, Box<Self>),
35//!     Sub(Box<Self>, Box<Self>),
36//!     Pow(Box<Self>, Box<Self>),
37//!     Neg(Box<Self>),
38//!     Factorial(Box<Self>),
39//!     Deref(Box<Self>),
40//!     Literal(i32),
41//! }
42//!
43//! impl std::fmt::Display for Expr {
44//!     fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
45//!         match self {
46//!             Self::Literal(literal) => write!(f, "{literal}"),
47//!             Self::Factorial(left) => write!(f, "({left}!)"),
48//!             Self::Deref(left) => write!(f, "(*{left})"),
49//!             Self::Neg(right) => write!(f, "(-{right})"),
50//!             Self::Add(left, right) => write!(f, "({left} + {right})"),
51//!             Self::Sub(left, right) => write!(f, "({left} - {right})"),
52//!             Self::Pow(left, right) => write!(f, "({left} ^ {right})"),
53//!         }
54//!     }
55//! }
56//!
57//! let atom = text::int::<_, extra::Err<Simple<char>>>(10)
58//!     .from_str()
59//!     .unwrapped()
60//!     .map(Expr::Literal)
61//!     .padded();
62//!
63//! let op = |c| just(c).padded();
64//!
65//! let expr = atom.pratt((
66//!     // We want factorial to happen before any negation, so we need its precedence to be higher than `Expr::Neg`.
67//!     postfix(4, op('!'), |lhs, _, _| Expr::Factorial(Box::new(lhs))),
68//!     // Just like in math, we want that if we write -x^2, our parser parses that as -(x^2), so we need it to have
69//!     // exponents bind tighter than our prefix operators.
70//!     infix(right(3), op('^'), |l, _, r, _| Expr::Pow(Box::new(l), Box::new(r))),
71//!     // Notice the conflict with our `Expr::Sub`. This will still parse correctly. We want negation to happen before
72//!     // `+` and `-`, so we set its precedence higher.
73//!     prefix(2, op('-'), |_, rhs, _| Expr::Neg(Box::new(rhs))),
74//!     prefix(2, op('*'), |_, rhs, _| Expr::Deref(Box::new(rhs))),
75//!     // Our `-` and `+` bind the weakest, meaning that even if they occur first in an expression, they will be the
76//!     // last executed.
77//!     infix(left(1), op('+'), |l, _, r, _| Expr::Add(Box::new(l), Box::new(r))),
78//!     infix(left(1), op('-'), |l, _, r, _| Expr::Sub(Box::new(l), Box::new(r))),
79//! ))
80//!     .map(|x| x.to_string());
81//!
82//! assert_eq!(
83//!     expr.parse("*1 + -2! - -3^2").into_result(),
84//!     Ok("(((*1) + (-(2!))) - (-(3 ^ 2)))".to_string()),
85//! );
86//! ```

437/// Defines the [associativity](https://en.wikipedia.org/wiki/Associative_property) and binding power of an [`infix`]
438/// operator (see [`left`] and [`right`]).
439///
440/// Higher binding powers should be used for higher precedence operators.
441#[derive(Copy, Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
442pub enum Associativity {
443    /// Specifies that the operator should be left-associative, with the given binding power (see [`left`]).
444    Left(u16),
445    /// Specifies that the operator should be right-associative, with the given binding power (see [`right`]).
446    Right(u16),
447}
448
449/// Specifies a left [`Associativity`] with the given binding power.
450///
451/// Left-associative operators are evaluated from the left-most terms, moving rightward. For example, the expression
452/// `a + b + c + d` will be evaluated as `((a + b) + c) + d` because addition is conventionally left-associative.
453pub fn left(binding_power: u16) -> Associativity {
454    Associativity::Left(binding_power)
455}
456
457/// Specifies a right [`Associativity`] with the given binding power.
458///
459/// Right-associative operators are evaluated from the right-most terms, moving leftward. For example, the expression
460/// `a ^ b ^ c ^ d` will be evaluated as `a ^ (b ^ (c ^ d))` because exponents are conventionally right-associative.
461pub fn right(binding_power: u16) -> Associativity {
462    Associativity::Right(binding_power)
463}
464
465impl Associativity {
466    fn left_power(&self) -> u32 {
467        match self {
468            Self::Left(x) => *x as u32 * 2,
469            Self::Right(x) => *x as u32 * 2 + 1,
470        }
471    }
472
473    fn right_power(&self) -> u32 {
474        match self {
475            Self::Left(x) => *x as u32 * 2 + 1,
476            Self::Right(x) => *x as u32 * 2,
477        }
478    }
479}

502/// Specify a binary infix operator for a pratt parser with the given associativity, binding power, and
503/// [fold function](crate::pratt#fold-functions).
504///
505/// Operators like addition, subtraction, multiplication, division, remainder, exponentiation, etc. are infix binary
506/// operators in most languages.
507///
508/// See [`left`] and [`right`] for information about associativity.
509///
510/// The fold function (the last argument) tells the parser how to combine the operator and operands into a new
511/// expression. It must have the following signature:
512///
513/// ```ignore
514/// impl Fn(Atom, Op, Atom, &mut MapExtra<'src, '_, I, E>) -> O
515/// ```
516pub const fn infix<'src, A, F, Atom, Op, I, E>(
517    associativity: Associativity,
518    op_parser: A,
519    fold: F,
520) -> Infix<'src, A, F, Atom, Op, I, E>
521where
522    F: Fn(Atom, Op, Atom, &mut MapExtra<'src, '_, I, E>) -> Atom,
523{
524    Infix {
525        op_parser,
526        fold,
527        associativity,
528        phantom: EmptyPhantom::new(),
529    }
530}

545        lhs: M::Output<O>,
546        min_power: u32,
547        f: &impl Fn(&mut InputRef<'src, 'parse, I, E>, u32) -> PResult<M, O>,
548    ) -> Result<M::Output<O>, M::Output<O>>
549    where
550        Self: Sized,
551    {
552        if self.associativity.left_power() >= min_power {
553            match self.op_parser.go::<M>(inp) {
554                Ok(op) => match f(inp, self.associativity.right_power()) {
555                    Ok(rhs) => Ok(M::combine(
556                        M::combine(lhs, rhs, |lhs, rhs| (lhs, rhs)),
557                        op,
558                        |(lhs, rhs), op| {
559                            (self.fold)(lhs, op, rhs, &mut MapExtra::new(pre_expr, inp))
560                        },
561                    )),
562                    Err(()) => {
563                        inp.rewind(pre_op.clone());
564                        Err(lhs)
565                    }
566                },
567                Err(()) => {
568                    inp.rewind(pre_op.clone());
569                    Err(lhs)
570                }
571            }
572        } else {
573            Err(lhs)
574        }
575    }
576
577    op_check_and_emit!();
578}

914#[allow(unused_variables, non_snake_case)]
915impl<'src, Atom, Ops> Pratt<Atom, Ops> {
916    #[inline]
917    fn pratt_go<M: Mode, I, O, E>(
918        &self,
919        inp: &mut InputRef<'src, '_, I, E>,
920        min_power: u32,
921    ) -> PResult<M, O>
922    where
923        I: Input<'src>,
924        E: ParserExtra<'src, I>,
925        Atom: Parser<'src, I, O, E>,
926        Ops: Operator<'src, I, O, E>,
927    {
928        let pre_expr = inp.save();
929        // Prefix unary operators
930        let mut lhs = match self
931            .ops
932            .do_parse_prefix::<M>(inp, &pre_expr, &|inp, min_power| {
933                recursive::recurse(|| self.pratt_go::<M, _, _, _>(inp, min_power))
934            }) {
935            Ok(out) => out,
936            Err(()) => self.atom.go::<M>(inp)?,
937        };
938
939        loop {
940            let pre_op = inp.save();
941
942            // Postfix unary operators
943            match self
944                .ops
945                .do_parse_postfix::<M>(inp, pre_expr.cursor(), &pre_op, lhs, min_power)
946            {
947                Ok(out) => {
948                    lhs = out;
949                    continue;
950                }
951                Err(out) => lhs = out,
952            }
953
954            // Infix binary operators
955            match self.ops.do_parse_infix::<M>(
956                inp,
957                pre_expr.cursor(),
958                &pre_op,
959                lhs,
960                min_power,
961                &|inp, min_power| {
962                    recursive::recurse(|| self.pratt_go::<M, _, _, _>(inp, min_power))
963                },
964            ) {
965                Ok(out) => {
966                    lhs = out;
967                    continue;
968                }
969                Err(out) => lhs = out,
970            }
971
972            inp.rewind(pre_op);
973            break;
974        }
975
976        Ok(lhs)
977    }
978}

1168    #[test]
1169    fn with_prefix_ops() {
1170        let atom = text::int::<_, Err<Simple<char>>>(10)
1171            .from_str()
1172            .unwrapped()
1173            .map(Expr::Literal);
1174
1175        let parser = atom
1176            .pratt((
1177                // -- Prefix
1178                // Because we defined '*' and '/' as right associative operators,
1179                // in order to get these to function as expected, their strength
1180                // must be higher
1181                prefix(2, just('-'), |_, r, _| u(Expr::Negate, r)),
1182                prefix(2, just('~'), |_, r, _| u(Expr::Not, r)),
1183                // This is what happens when not
1184                prefix(1, just('§'), |_, r, _| u(Expr::Confusion, r)),
1185                // -- Infix
1186                infix(left(0), just('+'), |l, _, r, _| i(Expr::Add, l, r)),
1187                infix(left(0), just('-'), |l, _, r, _| i(Expr::Sub, l, r)),
1188                infix(right(1), just('*'), |l, _, r, _| i(Expr::Mul, l, r)),
1189                infix(right(1), just('/'), |l, _, r, _| i(Expr::Div, l, r)),
1190            ))
1191            .map(|x| x.to_string());
1192

1194            parser.parse("-1+§~2*3").into_result(),
