17pub(crate) type PResult<M, O> = Result<<M as Mode>::Output<O>, ()>;
18/// The result of calling [`IterParser::next`]
19pub(crate) type IPResult<M, O> = Result<Option<<M as Mode>::Output<O>>, ()>;

21/// An abstract parse mode - can be [`Emit`] or [`Check`] in practice, and represents the
22/// common interface for handling both in the same method.
23pub trait Mode {
24    /// The output of this mode for a given type
25    type Output<T>;
26
27    /// Bind the result of a closure into an output
28    fn bind<T, F: FnOnce() -> T>(f: F) -> Self::Output<T>;
29
30    /// Given an [`Output`](Self::Output), takes its value and return a newly generated output
31    fn map<T, U, F: FnOnce(T) -> U>(x: Self::Output<T>, f: F) -> Self::Output<U>;
32
33    /// Choose between two fallible functions to execute depending on the mode.
34    fn choose<A, T, E, F: FnOnce(A) -> Result<T, E>, G: FnOnce(A) -> Result<(), E>>(
35        arg: A,
36        f: F,
37        g: G,
38    ) -> Result<Self::Output<T>, E>;
39
40    /// Given two [`Output`](Self::Output)s, take their values and combine them into a new
41    /// output value
42    fn combine<T, U, V, F: FnOnce(T, U) -> V>(
43        x: Self::Output<T>,
44        y: Self::Output<U>,
45        f: F,
46    ) -> Self::Output<V>;
47    /// By-reference version of [`Mode::combine`].
48    fn combine_mut<T, U, F: FnOnce(&mut T, U)>(x: &mut Self::Output<T>, y: Self::Output<U>, f: F);
49
50    /// Given an array of outputs, bind them into an output of arrays
51    fn array<T, const N: usize>(x: [Self::Output<T>; N]) -> Self::Output<[T; N]>;
52
53    fn from_mut<T>(r: &mut Self::Output<T>) -> Self::Output<&mut T>;
54
55    fn get_or<T, F: FnOnce() -> T>(r: Self::Output<T>, f: F) -> T;
56
57    /// Invoke a parser user the current mode. This is normally equivalent to
58    /// [`parser.go::<M>(inp)`](Parser::go), but it can be called on unsized values such as
59    /// `dyn Parser`.
60    fn invoke<'a, I, O, E, P>(parser: &P, inp: &mut InputRef<'a, '_, I, E>) -> PResult<Self, O>
61    where
62        I: Input<'a>,
63        E: ParserExtra<'a, I>,
64        P: Parser<'a, I, O, E> + ?Sized;
65
66    /// Invoke a parser with configuration using the current mode. This is normally equivalent
67    /// to [`parser.go::<M>(inp)`](ConfigParser::go_cfg), but it can be called on unsized values
68    /// such as `dyn Parser`.
69    fn invoke_cfg<'a, I, O, E, P>(
70        parser: &P,
71        inp: &mut InputRef<'a, '_, I, E>,
72        cfg: P::Config,
73    ) -> PResult<Self, O>
74    where
75        I: Input<'a>,
76        E: ParserExtra<'a, I>,
77        P: ConfigParser<'a, I, O, E> + ?Sized;
78
79    #[cfg(feature = "pratt")]
80    fn invoke_pratt_op_prefix<'src, 'parse, Op, I, O, E>(
81        op: &Op,
82        inp: &mut InputRef<'src, 'parse, I, E>,
83        pre_expr: &input::Checkpoint<'src, 'parse, I, <E::State as Inspector<'src, I>>::Checkpoint>,
84        f: &impl Fn(&mut InputRef<'src, 'parse, I, E>, u32) -> PResult<Self, O>,
85    ) -> PResult<Self, O>
86    where
87        Op: pratt::Operator<'src, I, O, E>,
88        I: Input<'src>,
89        E: ParserExtra<'src, I>;
90    #[cfg(feature = "pratt")]
91    fn invoke_pratt_op_postfix<'src, 'parse, Op, I, O, E>(
92        op: &Op,
93        inp: &mut InputRef<'src, 'parse, I, E>,
94        pre_expr: &input::Cursor<'src, 'parse, I>,
95        pre_op: &input::Checkpoint<'src, 'parse, I, <E::State as Inspector<'src, I>>::Checkpoint>,
96        lhs: Self::Output<O>,
97        min_power: u32,
98    ) -> Result<Self::Output<O>, Self::Output<O>>
99    where
100        Op: pratt::Operator<'src, I, O, E>,
101        I: Input<'src>,
102        E: ParserExtra<'src, I>;
103    #[cfg(feature = "pratt")]
104    fn invoke_pratt_op_infix<'src, 'parse, Op, I, O, E>(
105        op: &Op,
106        inp: &mut InputRef<'src, 'parse, I, E>,
107        pre_expr: &input::Cursor<'src, 'parse, I>,
108        pre_op: &input::Checkpoint<'src, 'parse, I, <E::State as Inspector<'src, I>>::Checkpoint>,
109        lhs: Self::Output<O>,
110        min_power: u32,
111        f: &impl Fn(&mut InputRef<'src, 'parse, I, E>, u32) -> PResult<Self, O>,
112    ) -> Result<Self::Output<O>, Self::Output<O>>
113    where
114        Op: pratt::Operator<'src, I, O, E>,
115        I: Input<'src>,
116        E: ParserExtra<'src, I>;
117}

119/// Emit mode - generates parser output
120pub struct Emit;
121
122impl Mode for Emit {
123    type Output<T> = T;
124    #[inline(always)]
125    fn bind<T, F: FnOnce() -> T>(f: F) -> Self::Output<T> {
126        f()
127    }
128    #[inline(always)]
129    fn map<T, U, F: FnOnce(T) -> U>(x: Self::Output<T>, f: F) -> Self::Output<U> {
130        f(x)
131    }
132    #[inline(always)]
133    fn choose<A, T, E, F: FnOnce(A) -> Result<T, E>, G: FnOnce(A) -> Result<(), E>>(
134        arg: A,
135        f: F,
136        _: G,
137    ) -> Result<Self::Output<T>, E> {
138        f(arg)
139    }
140    #[inline(always)]
141    fn combine<T, U, V, F: FnOnce(T, U) -> V>(
142        x: Self::Output<T>,
143        y: Self::Output<U>,
144        f: F,
145    ) -> Self::Output<V> {
146        f(x, y)
147    }
148    #[inline(always)]
149    fn combine_mut<T, U, F: FnOnce(&mut T, U)>(x: &mut Self::Output<T>, y: Self::Output<U>, f: F) {
150        f(x, y)
151    }
152    #[inline(always)]
153    fn array<T, const N: usize>(x: [Self::Output<T>; N]) -> Self::Output<[T; N]> {
154        x
155    }
156
157    #[inline(always)]
158    fn from_mut<T>(r: &mut Self::Output<T>) -> Self::Output<&mut T> {
159        r
160    }
161    #[inline(always)]
162    fn get_or<T, F: FnOnce() -> T>(r: Self::Output<T>, _f: F) -> T {
163        r
164    }
165
166    #[inline(always)]
167    fn invoke<'a, I, O, E, P>(parser: &P, inp: &mut InputRef<'a, '_, I, E>) -> PResult<Self, O>
168    where
169        I: Input<'a>,
170        E: ParserExtra<'a, I>,

175
176    #[inline(always)]
177    fn invoke_cfg<'a, I, O, E, P>(
178        parser: &P,
179        inp: &mut InputRef<'a, '_, I, E>,
180        cfg: P::Config,
181    ) -> PResult<Self, O>
182    where
183        I: Input<'a>,
184        E: ParserExtra<'a, I>,
185        P: ConfigParser<'a, I, O, E> + ?Sized,
186    {
187        parser.go_emit_cfg(inp, cfg)
188    }
189
190    #[cfg(feature = "pratt")]
191    #[inline(always)]
192    fn invoke_pratt_op_prefix<'src, 'parse, Op, I, O, E>(
193        op: &Op,
194        inp: &mut InputRef<'src, 'parse, I, E>,
195        pre_expr: &input::Checkpoint<'src, 'parse, I, <E::State as Inspector<'src, I>>::Checkpoint>,
196        f: &impl Fn(&mut InputRef<'src, 'parse, I, E>, u32) -> PResult<Self, O>,
197    ) -> PResult<Self, O>
198    where
199        Op: pratt::Operator<'src, I, O, E>,
200        I: Input<'src>,
201        E: ParserExtra<'src, I>,
202    {
203        op.do_parse_prefix_emit(inp, pre_expr, &f)
204    }
205    #[cfg(feature = "pratt")]
206    #[inline(always)]
207    fn invoke_pratt_op_postfix<'src, 'parse, Op, I, O, E>(
208        op: &Op,
209        inp: &mut InputRef<'src, 'parse, I, E>,
210        pre_expr: &input::Cursor<'src, 'parse, I>,
211        pre_op: &input::Checkpoint<'src, 'parse, I, <E::State as Inspector<'src, I>>::Checkpoint>,
212        lhs: Self::Output<O>,
213        min_power: u32,
214    ) -> Result<Self::Output<O>, Self::Output<O>>
215    where
216        Op: pratt::Operator<'src, I, O, E>,
217        I: Input<'src>,
218        E: ParserExtra<'src, I>,
219    {
220        op.do_parse_postfix_emit(inp, pre_expr, pre_op, lhs, min_power)
221    }
222    #[cfg(feature = "pratt")]
223    #[inline(always)]
224    fn invoke_pratt_op_infix<'src, 'parse, Op, I, O, E>(
225        op: &Op,
226        inp: &mut InputRef<'src, 'parse, I, E>,
227        pre_expr: &input::Cursor<'src, 'parse, I>,
228        pre_op: &input::Checkpoint<'src, 'parse, I, <E::State as Inspector<'src, I>>::Checkpoint>,
229        lhs: Self::Output<O>,
230        min_power: u32,
231        f: &impl Fn(&mut InputRef<'src, 'parse, I, E>, u32) -> PResult<Self, O>,
232    ) -> Result<Self::Output<O>, Self::Output<O>>
233    where
234        Op: pratt::Operator<'src, I, O, E>,
235        I: Input<'src>,

353// TODO: Remove this when MaybeUninit transforms to/from arrays stabilize in any form
354pub(crate) trait MaybeUninitExt<T>: Sized {
355    /// Identical to the unstable [`MaybeUninit::uninit_array`]
356    fn uninit_array<const N: usize>() -> [Self; N];
357
358    /// Identical to the unstable [`MaybeUninit::array_assume_init`]
359    unsafe fn array_assume_init<const N: usize>(uninit: [Self; N]) -> [T; N];
360}
361
362impl<T> MaybeUninitExt<T> for MaybeUninit<T> {
363    #[allow(clippy::uninit_assumed_init)]
364    fn uninit_array<const N: usize>() -> [Self; N] {
365        // SAFETY: Output type is entirely uninhabited - IE, it's made up entirely of `MaybeUninit`
366        unsafe { MaybeUninit::uninit().assume_init() }
367    }
368
369    unsafe fn array_assume_init<const N: usize>(uninit: [Self; N]) -> [T; N] {
370        (&uninit as *const [Self; N] as *const [T; N]).read()
371    }
372}
